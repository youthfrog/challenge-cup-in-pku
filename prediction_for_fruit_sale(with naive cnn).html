class fruitDataset(Dataset):
    def __init__(self, data_path, data_name):
        self.features = pickle.load(open(data_path+data_name+'.pkl','rb'))
        self.label = pickle.load(open(data_path+data_name+'_label'+'.pkl','rb'))
        #self.label = [x for x in open(data_path+data_name+'.txt','r', encoding='utf-8').readlines()]
    
    def __len__(self):
        return len(self.label)
    
    def __getitem__(self, index):
        data, label = self.features[index], self.label[index]
        return data,label
    
def collate_fn(batch):
    feature_list = []
    label_list = []
    for feature, label in batch:
        feature_list.append(torch.FloatTensor([float(item) for item in feature]))
        label_list.append(int(label[0]))
    labels = torch.Tensor(label_list)
    features = torch.cat(feature_list, dim=0)
    return features, labels

class Config:
    batch_size = 5
    hidden_size1 = 16
    hidden_size2 = 4
    input_len = 36
    

class MyModel(nn.Module):
    def __init__(self,config):
        super(MyModel, self).__init__()
        self.batch_size = config.batch_size
        self.input_size = config.input_len
        self.l1 = nn.Linear(self.input_size, config.hidden_size1)
        self.l2 = nn.Linear(config.hidden_size1, config.hidden_size2)
        self.l3 = nn.Linear(config.hidden_size2, 1)
        self.bn1 = nn.BatchNorm1d(config.hidden_size1)
        self.bn2 = nn.BatchNorm1d(config.hidden_size2)
        
        
    def forward(self, data):
        data = data.reshape(self.batch_size, -1)
        output = F.relu(self.l1(data))
        output = self.bn1(output)
        output = F.relu(self.l2(output))
        output = self.bn2(output)
        value = self.l3(output)
        return value
    
def train(epoch):
    model.train()
    iterations = 0
    loss_cum = 0
    
    for data, label in train_loader:
        data = data.to('cpu')
        label = label.to('cpu')
        iterations += 1
        output = model(data)
        output = output.squeeze(dim=-1)
        loss = criterion(output,label)
        loss.backward()
        nn.utils.clip_grad_norm_(model.parameters(), 1.0)
        if iterations % 1 == 0:
            optimizer.step()
            optimizer.zero_grad()
        loss_cum += loss.item()
    print('Epoch %i, train_loss = %f' % (epoch, loss_cum/iterations))
    return loss_cum/iterations

def eval(epoch):
    model.eval()
    count, loss = 0, 100
    with torch.no_grad():
        for data, label in valid_loader:
            data = data.to('cpu')
            label = label.to('cpu')
            output = model(data).squeeze(dim=-1)
            # print(label, output)
            tmp = ((output-label)*(output-label)).mean()
            loss = (loss*count + tmp.item())/(count+1)
            count += 1
        print('valid loss = ', loss)
    return loss
